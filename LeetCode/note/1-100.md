## 2.两数相加(中等)

```c
struct ListNode* addTwoNumbers(struct ListNode* l1, struct ListNode* l2) {
    struct ListNode *head, *ptr;
    int next_bit;

    head = NULL;

    while(l1 || l2) {
        if(l1 && l2) {
            if(!head) {
                head = (struct ListNode*)malloc(sizeof(struct ListNode));
                head->next = NULL;

                head->val = (l1->val + l2->val) % 10;
                next_bit = (l1->val + l2->val) / 10;

                ptr = head;
            }
            else {
                ptr->next = (struct ListNode*)malloc(sizeof(struct ListNode));
                ptr = ptr->next;

                ptr->val = (l1->val + l2->val + next_bit) % 10;
                next_bit = (l1->val + l2->val + next_bit) / 10;
                ptr->next = NULL;
            }

            l1 = l1->next;
            l2 = l2->next;
        }
        else if(l1) {
            ptr->next = (struct ListNode*)malloc(sizeof(struct ListNode));
            ptr = ptr->next;

            ptr->val = (l1->val + next_bit) % 10;
            next_bit = (l1->val + next_bit) / 10;
            ptr->next = NULL;

            l1 = l1->next;
        }
        else {
            ptr->next = (struct ListNode*)malloc(sizeof(struct ListNode));
            ptr = ptr->next;

            ptr->val = (l2->val + next_bit) % 10;
            next_bit = (l2->val + next_bit) / 10;
            ptr->next = NULL;

            l2 = l2->next;
        }
    }

    if(next_bit > 0) {
        ptr->next = (struct ListNode*)malloc(sizeof(struct ListNode));
        ptr = ptr->next;

        ptr->val = next_bit;
        ptr->next = NULL;
    }

    return head;
}
```

```c
bool isPalindrome(int x) {
    bool retBoolean;
    unsigned int x_copy, x_reverse;
    
    if(x < 0) {
        retBoolean = false;
    }
    else if(x < 10) {
        retBoolean = true;
    }
    else {
        x_copy = x;
        x_reverse = 0;

        while(x) {
            x_reverse = x_reverse * 10 + x % 10;

        }

        retBoolean = (x_copy == x_reverse);
    }

    return retBoolean;
```

## 11.盛最多水的容器(中等)

```c
int maxArea(int* height, int heightSize) {
    int left, right, max_area, h, area;

    left = 0;
    right = heightSize - 1;

    max_area = 0;

    while(left < right) {
        h = height[left] < height[right] ? height[left] : height[right];

        area = (right - left) * h;

        max_area = area > max_area ? area : max_area;

        height[left] < height[right] ? left++ : right--;
    }

    return max_area;
}
```

## 20.有效的括号(简单)

https://leetcode.cn/problems/valid-parentheses/description/

```c
# define MAXN 10000

struct SNode {
    char SArr[MAXN];
    int top;
};
typedef struct SNode *Stack;

Stack makeStack() {
    Stack s;

    s = (Stack)malloc(sizeof(struct SNode));
    s->top = -1;

    return s;
}

void push(Stack s, char bracket) {
    (s->SArr)[++(s->top)] = bracket;
}

char pop(Stack s) {
    return (s->SArr)[(s->top)--];
}

bool isEmptyStack(Stack s) {
    return s->top < 0;
}

bool isValid(char * s){
    bool retBool;
    Stack stack;
    char bracket;

    stack = makeStack();

    retBool = true;

    for(int i = 0; s[i]; i++) {
        if(s[i] == '(' || s[i] == '[' || s[i] == '{') {
            push(stack, s[i]);
        }
        else {
            if(isEmptyStack(stack)) {
                retBool = false;
                break;
            }
            else {
                bracket = pop(stack);

                if(bracket == '(' && s[i] != ')') {
                    retBool = false;
                    break;
                }
                else if(bracket == '[' && s[i] != ']') {
                    retBool = false;
                    break;
                }
                else if(bracket == '{' && s[i] != '}') {
                    retBool = false;
                    break;
                }
            }
        }
    }

    if(!isEmptyStack(stack)) {
        retBool = false;
    }

    return retBool;
}
```

```c
struct ListNode* mergeTwoLists(struct ListNode* list1, struct ListNode* list2){
    struct ListNode *head, *ptr;

    if(list1 == NULL) {
        head = list2;
    }
    else if(list2 == NULL) {
        head = list1;
    }
    else {
        if(list1->val <= list2->val) {
            head = list1;
            list1 = list1->next;
        }
        else {
            head = list2;
            list2 = list2->next;
        }
        head->next = NULL;
        ptr = head;

        while(list1 && list2) {
            if(list1->val <= list2->val) {
                ptr->next = list1;
                list1 = list1->next;
            }
            else {
                ptr->next = list2;
                list2 = list2->next;
            }
            ptr = ptr->next;
            ptr->next = NULL;
        }

        ptr->next = list1 ? list1 : list2;
    }

    return head;
}
```

```c
struct ListNode* swapPairs(struct ListNode* head) {

    new_head = head;

    if(head && head->next) {
        q = head;
        p = head->next;
        new_head = p;

        tmp = p->next;
        p->next = q;
        q->next = tmp;

        while(tmp && tmp->next) {
            r = q;
            q = r->next;
            p = q->next;

            p->next = q;
            r->next = p;
            q->next = tmp;
        }
    }

    return new_head;
```

## 26.删除有序数组中的重复项(简单)

https://leetcode.cn/problems/remove-duplicates-from-sorted-array/description/

### 解法1：暴力求解

```c
int removeDuplicates(int* nums, int numsSize) {
    for(int i = 1; i < numsSize; i++) {
        if(nums[i] == nums[i-1]) {
            for(int j = i; j < numsSize-1; j++) {
                nums[j] = nums[j+1];
            }
            numsSize -= 1;
            i -= 1;
        }
    }

    return numsSize;
}
```

### 解法2：双指针法

```c
int removeDuplicates(int* nums, int numsSize) {
    int fast_ptr, slow_ptr;

    slow_ptr = fast_ptr = 1;

    while(fast_ptr < numsSize) {
        if(nums[fast_ptr] != nums[fast_ptr-1]) {
            nums[slow_ptr++] = nums[fast_ptr];
        }

        fast_ptr += 1;
    }

    return slow_ptr;
}
```

双指针一块一慢，保证一次遍历就可以实现，时间复杂度为$O(n)$.

```c
int strStr(char * haystack, char * needle) {
    int slow_ptr, fast_ptr, in_needle;

    if(strlen(haystack) < strlen(needle))
        return -1;
    else if(strlen(needle) == 1) {
        for(int i = 0; i < strlen(haystack); i++)
        {
            if(haystack[i] == needle[0])
                return i;
        }

        return -1;
    }

    slow_ptr = fast_ptr = 0;
    in_needle = 0;

    while(fast_ptr < strlen(haystack)) {
        if(in_needle) {
            if(haystack[fast_ptr] == needle[fast_ptr-slow_ptr] && haystack[fast_ptr] != '\0')
                fast_ptr += 1;
            else {
                slow_ptr += 1;
                fast_ptr = slow_ptr;
                in_needle = 0;
            }
        }
        else {
            if(haystack[fast_ptr] == needle[0])
                in_needle = 1;
            else
                slow_ptr = ++fast_ptr;
        }

        if(haystack[fast_ptr] == needle[fast_ptr-slow_ptr] && fast_ptr-slow_ptr+1 == strlen(needle))
            break;

        fast_ptr -= 1;
    }
```

## 53.最大子数组和(中等)

https://leetcode.cn/problems/maximum-subarray/description/

### 解法1：在线算法

```c
int maxSubArray(int* nums, int numsSize){
    int max_sum, sub_sum;
    bool is_all_neg;

    is_all_neg = true;

    max_sum = sub_sum = 0;

    for(int i = 0; i < numsSize; i++) {
        if(nums[i] >= 0)
            is_all_neg = false;

        sub_sum += nums[i];

        if(sub_sum > max_sum)
            max_sum = sub_sum;
        else if(sub_sum < 0)
            sub_sum = 0;
    }

    if(is_all_neg) {
        max_sum = nums[0];

        for(int i = 1; i < numsSize; i++) {
            if(nums[i] > max_sum)
                max_sum = nums[i];
        }
    }

    return max_sum;
}
```

最快的算法，时间复杂度为$O(n)$.

### 解法2：分治算法

```c
#define MIN -10000

int maxSubArray(int* nums, int numsSize) {
    int *left_sub_arr, *right_sub_arr, left_sub_max, right_sub_max;
    int left, right, mid, left_sum, left_max_sum, right_sum, right_max_sum;
    int retVal;

    if(numsSize <= 0) {
        return MIN;
    }
    else if(numsSize == 1) {
        return nums[0];
    }

    left = 0;
    right = numsSize - 1;
    mid = (left + right) / 2;

    left_sub_arr = (int*)malloc(sizeof(int)*(mid-left+1));
    right_sub_arr = (int*)malloc(sizeof(int)*(right-mid));

    for(int i = 0; i <= mid; i++) {
        left_sub_arr[i] = nums[i];
    }
    for(int i = 0; i < right-mid; i++) {
        right_sub_arr[i] = nums[mid+1+i];
    }

    left_sub_max = maxSubArray(left_sub_arr, mid-left+1);
    right_sub_max = maxSubArray(right_sub_arr, right-mid);

    left_max_sum = right_max_sum = MIN;
    left_sum = right_sum = 0;

    for(int i = mid; i >= left; i--) {
        left_sum += nums[i];

        if(left_sum > left_max_sum) {
            left_max_sum = left_sum;
        }
    }

    for(int i = mid+1; i <= right; i++) {
        right_sum += nums[i];

        if(right_sum > right_max_sum) {
            right_max_sum = right_sum;
        }
    }

    if(left_sub_max >= right_sub_max && left_sub_max >= right_max_sum+left_max_sum) {
        retVal = left_sub_max;
    }
    else if(right_sub_max >= left_sub_max && right_sub_max >= right_max_sum+left_max_sum) {
        retVal = right_sub_max;
    }
    else {
        retVal = left_max_sum + right_max_sum;
    }

    free(left_sub_arr);
    free(right_sub_arr);
    
    return retVal;
}
```

## 94.二叉树的中序遍历(简单)

https://leetcode.cn/problems/binary-tree-inorder-traversal/

### 解法1：递归实现

```c
int* inorderTraversal(struct TreeNode* root, int* returnSize)
{
    int left_size, right_size, *left_sub_tree, *right_sub_tree, *retArr, index;

    if(root == NULL) {
        *returnSize = 0;
        return NULL;
    } 
    else {
        left_sub_tree = inorderTraversal(root->left, &left_size);
        right_sub_tree = inorderTraversal(root->right, &right_size);
        
        *returnSize = 1 + left_size + right_size;
        retArr = (int*)malloc(sizeof(int)*(*returnSize));
        
        for(index = 0; index < left_size; index++)
            retArr[index] = left_sub_tree[index];
        retArr[index++] = root -> val;
        for(int i = 0; i < right_size; i++)
            retArr[index+i] = right_sub_tree[i];

        return retArr;
    }
}
```

### 解法2：迭代实现

借助栈和一个tag实现迭代遍历。

```c
#define MAXN 100

struct RNode {
    struct TreeNode *root;
    int is_first;
};
typedef struct RNode *PtrToRNode;

struct SNode {
    PtrToRNode SArr[MAXN];
    int top;
};
typedef struct SNode *Stack;

Stack makeStack() {
    Stack new_stack;

    new_stack = (Stack)malloc(sizeof(struct SNode));

    new_stack->top = -1;

    return new_stack;
}

void pushStack(Stack s, PtrToRNode ptr) {
    s->SArr[++s->top] = ptr;
}

PtrToRNode popStack(Stack s) {
    return s->SArr[s->top--];
}

int isEmptyStack(Stack s) {
    if(s->top == -1)
        return 1;
    else
        return 0;
}

int* inorderTraversal(struct TreeNode* root, int* returnSize) {
    Stack s;
    PtrToRNode root_node, new_root_node;
    int *retArr;

    retArr = NULL;
    *returnSize = 0;

    if(root) {
        *returnSize = 0;
        retArr = (int*)malloc(sizeof(int)*100);

        s = makeStack();

        if(root->right) {
            root_node = (PtrToRNode)malloc(sizeof(struct RNode));
            root_node->root = root->right;
            root_node->is_first = 1;

            pushStack(s, root_node);
        }
        
        root_node = (PtrToRNode)malloc(sizeof(struct RNode));
        root_node->root = root;
        root_node->is_first = 0;
        pushStack(s, root_node);

        if(root->left) {
            root_node = (PtrToRNode)malloc(sizeof(struct RNode));
            root_node->root = root->left;
            root_node->is_first = 1;

            pushStack(s, root_node);
        }

        while(!isEmptyStack(s)) {
            root_node = popStack(s);

            if(root_node->is_first == 0) {
                retArr[(*returnSize)++] = root_node->root->val;

                free(root_node);
            }
            else {
                if(root_node->root->right) {
                    new_root_node = (PtrToRNode)malloc(sizeof(struct RNode));
                    new_root_node->root = root_node->root->right;
                    new_root_node->is_first = 1;

                    pushStack(s, new_root_node);
                }

                root_node->is_first = 0;
                pushStack(s, root_node);

                if(root_node->root->left) {
                    new_root_node = (PtrToRNode)malloc(sizeof(struct RNode));
                    new_root_node->root = root_node->root->left;
                    new_root_node->is_first = 1;

                    pushStack(s, new_root_node);
                }
            }
        }
    }

    return retArr;
}
```



## 98.验证二叉搜索树(中等)

```c
int getMax(struct TreeNode* root) {
    struct TreeNode *ptr;
    int max_val;

    ptr = root;

    while(ptr) {
        max_val = ptr->val;
        ptr = ptr->right;
    }

    return max_val;
}

int getMin(struct TreeNode *root) {
    struct TreeNode *ptr;
    int min_val;

    ptr = root;

    while(ptr) {
        min_val = ptr->val;
        ptr = ptr->left;
    }

    return min_val;
}

bool isValidBST(struct TreeNode* root) {
    bool retBool;

    retBool = true;

    if(root) {
        if(!(isValidBST(root->left) && isValidBST(root->right))) {
            retBool = false;
        }
        else {
            if(root->left && root->right && !(root->val > getMax(root->left) && root->val < getMin(root->right)))
                retBool = false;
            else if(root->left && getMax(root->left) >= root->val)
                retBool = false;
            else if(root->right && getMin(root->right) <= root->val)
                retBool = false;
        }
    }

    return retBool;
}
```
