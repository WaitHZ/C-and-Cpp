## 424.替换后的最长重复字符(中等)

```c
int HashFunc(char c) {
    return c - 'A';
}

int characterReplacement(char * s, int k){
    int HashMap[26], retVal, len, left, right, max_count;

    if(strlen(s) <= k) {
        retVal = strlen(s);
    }
    else if(k == 0) {
        HashMap[0] = s[0];
        len = 1;
        retVal = 1;
        for(int i = 1; s[i]; i++) {
            if(s[i] == HashMap[0]) {
                len += 1;
                if(len > retVal) {
                    retVal = len;
                }
            }
            else {
                HashMap[0] = s[i];
                len = 1;
            }
        }
    }
    else {
        memset(HashMap, 0, sizeof(int)*26);
        max_count = 1;
        left = 0; right = 1;
        HashMap[HashFunc(s[0])] += 1;
        len = 1; 
        retVal = 1;
        while(s[right]) {
            if(right-left-max_count < k) {
                len = right - left + 1;
                HashMap[HashFunc(s[right])] += 1;
                if(len > retVal) {
                    retVal = len;
                }
                if(HashMap[HashFunc(s[right])] > max_count) {
                    max_count = HashMap[HashFunc(s[right])];
                }
                right += 1;
            }
            else {
                if(HashMap[HashFunc(s[right])] == max_count) {
                    len = right - left + 1;
                    HashMap[HashFunc(s[right])] += 1;
                    max_count += 1;
                    if(len > retVal) {
                        retVal = len;
                    }
                    right += 1;
                }
                else {
                    HashMap[HashFunc(s[left])] -= 1;
                    left += 1;
                    max_count = 0;
                    for(int i = 0; i < 26; i++) {
                        if(HashMap[i] > max_count) {
                            max_count = HashMap[i];
                        }
                    }
                }
            }
        }
    }

    return retVal;
}
```



## 494.目标和(中等)

https://leetcode.cn/problems/target-sum/description/

```c
void dfs(int *nums, int numsSize, int target, int *count, int sum) {
    
    
    if(numsSize == 1) {
        if(sum+nums[0] == target) {
            *count += 1;
        }
        if(sum-nums[0] == target) {
            *count += 1;
        }
    }
    else {
        dfs(&nums[1], numsSize-1, target, count, sum+nums[0]);
        dfs(&nums[1], numsSize-1, target, count, sum-nums[0]);
    }
}

int findTargetSumWays(int* nums, int numsSize, int target){
    int count;

    count = 0;

    dfs(nums, numsSize, target, &count, 0);

    return count;
}
```
