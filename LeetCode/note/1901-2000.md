## 1945.字符串转化后的各位数字之和(简单)

https://leetcode.cn/problems/sum-of-digits-of-string-after-convert/description/

```c
char *numToStr(int num) {
    int count, num_copy;
    char *str;

    num_copy = num;

    count = 0;
    while(num) {
        count += 1;
        num /= 10;
    }

    str = (char*)malloc(sizeof(char)*(count+1));
    str[count--] = '\0';

    while(num_copy) {
        str[count--] = '0' + num_copy % 10;
        num_copy /= 10;
    }

    return str;
}

int getLucky(char * s, int k){
    int num, index;
    char replaceStr[201];

    replaceStr[0] = '\0';

    index = 0;
    for(int i = 0; s[i]; i++) {
        strcat(replaceStr, numToStr(s[i]-'a'+1));
    }

    for(int i = 0; i < k; i++) {
        num = 0;
        for(int j = 0; replaceStr[j]; j++) {
            num += replaceStr[j] - '0';
        }
        strcpy(replaceStr, numToStr(num));
    }

    return num;
}
```
